var a = 1, b = 1, c, d;
c = ++a; alert(c); // 2
d = b++; alert(d); // 1
c = (2+ ++a); alert(c); // 5
d = (2+ b++); alert(d); // 4
alert(a); // 3
alert(b); // 3
Почему код даёт именно такие результаты?

Код Результат Комментарий
var a = 1, b = 1, c, d;
c = ++a; alert(c); 2 Префиксная форма инремента, a == 2 => c == 2
d = b++; alert(d); 1 Постфиксная форма инремента, b == 1 => d == 1
c = (2+ ++a); alert(c); 5 Префиксная форма инремента, a == 3 => c == 5
d = (2+ b++); alert(d); 4 Постфиксная форма инремента, b == 2 => d == 4
alert(a); 3 Из вычислений выше
alert(b); 3 Из вычислений выше
